CREATE TABLE IF NOT EXISTS keeper_matches (
    keeper_match VARCHAR(19) NOT NULL,
    player_match VARCHAR(17),
    pid VARCHAR(8),
    match_id VARCHAR(8),
    tls_id VARCHAR(17),
    team_match VARCHAR(17),
    season_long VARCHAR(12),
    date DATE,
    start VARCHAR(1),
    position VARCHAR(3),
    min INT,
    sot_against INT,
    goals_against INT,
    saves INT,
    save_pct REAL,
    clean_sheets INT,
    post_sh_xg REAL,
    pk_att_against INT,
    pk_allowed INT,
    pk_saved INT,
    pk_missed INT,
    launched_cmp INT,
    launched_att INT,
    launched_cmp_pct REAL,
    pass_att INT,
    throws_att INT,
    pass_launched_pct REAL,
    pass_avg_len REAL,
    gk_att INT,
    gk_launch_pct REAL,
    gk_avg_len REAL,
    crosses_faced INT,
    crosses_stopped INT,
    cross_stop_pct REAL,
    def_act_outside_pen INT,
    daop_avg_dist REAL,
    PRIMARY KEY (keeper_match),
    FOREIGN KEY (pid) REFERENCES players(pid),
    FOREIGN KEY (match_id) REFERENCES matches(match_id),
    FOREIGN KEY (tls_id) REFERENCES team_league_seasons(tls_id),
    FOREIGN KEY (team_match) REFERENCES team_matches(team_match)
);
