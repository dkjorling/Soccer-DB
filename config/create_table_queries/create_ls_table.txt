CREATE TABLE IF NOT EXISTS ls (
    ls_id TEXT NOT NULL,
    lg_id INTEGER,
    season INTEGER,
    season_long TEXT,
    lg_name TEXT,
    has_adv_stats TEXT,
    ls_api TEXT,
    has_rounds REAL,
    lg_start DATE,
    lg_end DATE,
    PRIMARY KEY (ls_id),
    FOREIGN KEY (lg_id) REFERENCES leagues(lg_id),
    FOREIGN KEY (season) REFERENCES seasons(season)
);
