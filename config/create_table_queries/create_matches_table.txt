CREATE TABLE IF NOT EXISTS matches (
    match_id VARCHAR(8) NOT NULL,
    date DATE,
    home_team_id VARCHAR(8),
    away_team_id VARCHAR(8),
    ls_id VARCHAR(15),
    PRIMARY KEY (match_id),
    FOREIGN KEY (ls_id) REFERENCES league_seasons(ls_id),
    FOREIGN KEY (home_team_id) REFERENCES teams(tid),
    FOREIGN KEY (away_team_id) REFERENCES teams(tid)
);
